# New Heir (Prince/Princess)
country_event = {
	id = GOT_SK_SYSTEM.2
	title = GOT_SK_SYSTEM.2.title
	desc = {
		trigger = {
			OR = {
				owns_or_subject_of = 221
				221 = {
					owner = {
						is_part_of_hre = yes
					}
				}
			}
		}
		desc = GOT_SK_SYSTEM.2.desc.a
	}
	desc = {
		trigger = {
			NOT = {
				OR = {
					owns_or_subject_of = 221
					221 = {
						owner = {
							is_part_of_hre = yes
						}
					}
				}
			}
		}
		desc = GOT_SK_SYSTEM.2.desc.b
	}
	picture = NEW_HEIR_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		is_emperor = yes
		has_regency = no
	}
	
	option = {
		name = GOT_SK_SYSTEM.2.option.a
		trigger = {
			OR = {
				owns_or_subject_of = 221
				221 = {
					owner = {
						is_part_of_hre = yes
					}
				}
			}
		}
		add_ruler_modifier = {
			name = got_prince_of_dragonstone
			duration = -1
			desc = ascension_ruler_desc
		}
		if = { # Dragonstone exists and is a Castellany
			limit = {
				A39 = {
					owns = 221 # Dragonstone
					is_part_of_hre = yes
					is_subject_of_type_with_overlord = {
						who = ROOT
						type = castellany
					}
				}
			}
			custom_tooltip = GOT_SK_SYSTEM.2.CAS1
			add_legitimacy = 10
			add_heir_claim = 20
		}
		else_if = { # Dragonstone exists but isn't a Castellany
			limit = {
				A39 = {
					owns = 221
					is_part_of_hre = yes
					NOT = {
						is_subject_of_type_with_overlord = {
							who = ROOT
							type = castellany
						}
					}
				}
			}
			custom_tooltip = GOT_SK_SYSTEM.2.CAS2
			hidden_effect = {
				A39 = {
					change_government = republic
					add_government_reform = castellany_of_dragonstone_reform
					set_country_flag = prince_of_dragonstone_flag
				}
				create_subject = {
					subject_type = castellany
					subject = A39
				}
			}
		}
		else_if = { # Dragonstone is owned and controlled by us
			limit = {
				owns = 221
			}
			if = {
				limit = {
					A39 = { exists = no }
				}
				custom_tooltip = GOT_SK_SYSTEM.2.CAS3a
				hidden_effect = {
					221 = { cede_province = A39 }
					A39 = {
						change_government = republic
						add_government_reform = castellany_of_dragonstone_reform
						set_country_flag = prince_of_dragonstone_flag
					}
					create_subject = {
						subject_type = castellany
						subject = A39
					}
				}
			}
			else = {
				221 = {
					add_province_modifier = {
						name = got_local_castellany
						duration = -1
					}
				}
				custom_tooltip = GOT_SK_SYSTEM.2.CAS3b
			}
		}
		else_if = { # Dragonstone is ruled by a Westeros lord or our subject
			limit = {
				221 = {
					owner = {
						NOT = { tag = A39 }
						OR = {
							is_part_of_hre = yes
							is_subject_of = ROOT
						}
					}
				}
			}
			if = {
				limit = {
					A39 = { exists = no }
				}
				custom_tooltip = GOT_SK_SYSTEM.2.CAS4
				hidden_effect = {
					221 = { cede_province = A39 }
					A39 = {
						change_government = republic
						add_government_reform = castellany_of_dragonstone_reform
						set_country_flag = prince_of_dragonstone_flag
					}
					create_subject = {
						subject_type = castellany
						subject = A39
					}
				}
			}
			else = {
				221 = {
					cede_province = ROOT
					add_province_modifier = {
						name = got_local_castellany
						duration = -1
					}
				}
				custom_tooltip = GOT_SK_SYSTEM.2.CAS3b 
			}
		}
	}

	option = {
		name = GOT_SK_SYSTEM.2.option.b
		trigger = {
			NOT = {
				OR = {
					owns_or_subject_of = 221
					221 = {
						owner = {
							is_part_of_hre = yes
						}
					}
				}
			}
		}
		add_ruler_modifier = {
			name = got_prince_of_westeros
			duration = -1
			desc = ascension_ruler_desc
		}
		custom_tooltip = GOT_SK_SYSTEM.2.CAS5 # Dragonstone is not ruled by us or a subject
	}
}

	}

		grant_independence = yes # Free them automatically
	}
	
	option = {
		name = GOT_SK_SYSTEM.30.option.a
		
		event_target:IronThrone = {
			create_subject = {
				subject_type = royal_subject

			}
		}
		
		ai_chance = {
			factor = 90
			
			modifier = {
				factor = 0.5

					}

			}
		}
	}
	option = {
		name = GOT_SK_SYSTEM.30.option.b
		
		event_target:IronThrone = {

			}
		}
		
		ai_chance = {
			factor = 10
			
			modifier = {
				factor = 0
				
				has_opinion = {
					who = event_target:IronThrone
					value = 25
				}
			}
		}
	}
}

# Claim the Iron Throne
country_event = {
	id = GOT_SK_SYSTEM.31
	title = GOT_SK_SYSTEM.31.title
	desc = GOT_SK_SYSTEM.31.desc
	picture = GOOD_WITH_MONARCH_eventPicture
	
	is_triggered_only = yes
	major = yes

	option = {
		name = GOT_SK_SYSTEM.31.option.a
	
		hidden_effect = {
			set_country_flag = iron_throne_transition
			save_event_target_as = originalCOU
		}
		
		# Moving over ruler to The Crownlands
		switch_tag_with_ruler = {
			ruler_string = King_of_Andals
			consort_string = Queen_of_Andals
			heir_string = Heir_of_Andals
			new_tag = CRW
		}
		
		save_global_event_target_as = IronThrone
		create_union = event_target:originalCOU

		set_emperor = yes
		add_prestige = 20
		add_dip_power = 50
		add_mil_power = 50
		add_adm_power = 50

		# Peace out all wars to stop glitchiness
		every_country = {
			limit = {
				war_with = ROOT
			}
			white_peace = ROOT
		}
		
		# Allow all current subjects to choose whether they become subjects of new holder
		every_country = {
			limit = {
				is_subject_of_type = royal_subject
			}
			country_event = { id = GOT_SK_SYSTEM.30 }
		}
		
		ai_chance = {
			factor = 100
		}
	}
	
	option = {
		name = GOT_SK_SYSTEM.31.option.b
		ai_chance = {
			factor = 100
			
			modifier = {
				factor = 0
				
				has_reform = noble_house_reform
			}
		}
		
		dismantle_empire_of_china = yes
		add_prestige = 100
		add_dip_power = 200
		add_mil_power = 200
		add_adm_power = 200
		
		every_country = {
			limit = {
				is_subject = yes
				is_subject_of_type = tributary_state
			}
			grant_independence = yes
		}
	}
}
