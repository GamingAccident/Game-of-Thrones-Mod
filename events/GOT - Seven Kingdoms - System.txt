# The Passing of the Throne
country_event = {
	id = GOT_SK_SYSTEM.30
	title = GOT_SK_SYSTEM.30.title
	desc = GOT_SK_SYSTEM.30.desc
	picture = GOOD_WITH_MONARCH_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		is_subject_of_type = royal_subject
	}
	
	immediate = {
		grant_independence = yes # Free them automatically
	}
	
	option = {
		name = GOT_SK_SYSTEM.30.option.a
		
		event_target:IronThroneHolder = {
			create_subject = {
				subject_type = royal_subject
				subject = ROOT
			}
		}
		
		ai_chance = {
			factor = 90
			
			modifier = {
				factor = 0.5
				
				NOT = {
					has_opinion = {
						who = event_target:IronThroneHolder
						value = -50
					}
				}
			}
		}
	}
	option = {
		name = GOT_SK_SYSTEM.30.option.b
		
		event_target:IronThroneHolder = {
			add_casus_belli = {
				target = ROOT
				type = cb_royal_conquest
			}
		}
		
		ai_chance = {
			factor = 10
			
			modifier = {
				factor = 0
				
				has_opinion = {
					who = event_target:IronThroneHolder
					value = 25
				}
			}
		}
	}
}

# Claim the Iron Throne
country_event = {
	id = GOT_SK_SYSTEM.31
	title = GOT_SK_SYSTEM.31.title
	desc = GOT_SK_SYSTEM.31.desc
	picture = GOOD_WITH_MONARCH_eventPicture
	
	is_triggered_only = yes
	major = yes

	option = {
		name = GOT_SK_SYSTEM.31.option.a
	
		hidden_effect = {
			set_country_flag = iron_throne_transition
			save_event_target_as = originalCOU
		}
		
		# Moving over ruler to The Crownlands
		switch_tag_with_ruler = {
			ruler_string = King_of_Andals
			consort_string = Queen_of_Andals
			heir_string = Heir_of_Andals
			new_tag = CRW
		}
		
		save_global_event_target_as = IronThroneHolder
		create_union = event_target:originalCOU

		set_emperor = yes
		add_prestige = 20
		add_dip_power = 50
		add_mil_power = 50
		add_adm_power = 50

		# Peace out all wars to stop glitchiness
		every_country = {
			limit = {
				war_with = ROOT
			}
			white_peace = ROOT
		}
		
		# Allow all current subjects to choose whether they become subjects of new holder
		every_country = {
			limit = {
				is_subject_of_type = royal_subject
			}
			country_event = { id = GOT_SK_SYSTEM.30 }
		}
		
		ai_chance = {
			factor = 100
		}
	}
	
	option = {
		name = GOT_SK_SYSTEM.31.option.b
		ai_chance = {
			factor = 100
			
			modifier = {
				factor = 0
				
				has_reform = noble_house_reform
			}
		}
		
		dismantle_empire_of_china = yes
		add_prestige = 100
		add_dip_power = 200
		add_mil_power = 200
		add_adm_power = 200
		
		every_country = {
			limit = {
				is_subject = yes
				is_subject_of_type = tributary_state
			}
			grant_independence = yes
		}
	}
}
