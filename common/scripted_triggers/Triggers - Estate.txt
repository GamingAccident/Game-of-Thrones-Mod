has_any_estates = {
	OR = {
		has_estate = estate_clergy
		has_estate = estate_merchants
		has_estate = estate_nobles
		has_estate = estate_commoners
		has_estate = estate_tribes
		has_estate = estate_adventurers
	}
}

# Used to determine if the country is strong enough to force 
# out the rest of the estates for actions that would have them
# removed.
crown_stronger_than_estate = {
	crown_land_share = $estate$
	estate_loyalty = {
		estate = $estate$
		higher_than_influence = yes
	}
}

has_clergy_trade_good_trigger = {
	OR = {
		trade_goods = wine
		trade_goods = wool
        trade_goods = tea
        trade_goods = coffee
        trade_goods = cotton
        trade_goods = candles
	}
}

estate_has_special_privileges_active = {
	# estate_privilege
	# estate_scope
	OR = {
		NOT = { has_estate_privilege = $estate_privilege$ }
		custom_trigger_tooltip = {
			tooltip = got_msq_active_$estate_scope$_reform
			has_government_attribute = estate_special_privileges
		}
	}
}

estate_special_privilege_check = {
	# estate_scope
	custom_trigger_tooltip = {
		tooltip = got_msq_active_$estate_scope$_reform
		has_government_attribute = estate_special_privileges
	}
}
