ages_effect_bonus = {
	add_country_modifier = {
		name = ages_$bonus_name$_modifier
		duration = -1
	}
	custom_tooltip = "ages_effect_tt"
}

new_age_on_action_effect = {
	if = {
		limit = {
			current_age = age_of_fire
		}
		age_of_fire_nation_bonus_effect = yes
	}
	else_if = {
		limit = {
			current_age = age_of_rebellion
		}
		age_of_rebellion_nation_bonus_effect = yes
	}
}
